# 解法一

使用变形的BFS

```javascript
var levelOrder = function (root) {
    //对于特殊数据的处理
    if(!root) return []
    //定义结果字段
    let res = []
    //定义一个队列，用来放置每一层的节点
    let queue = [root]

    while (queue.length) {
        //获取当前层的长度
        let length = queue.length
        //用于存放当前层的所有值
        let level = []
        //遍历当前层
        for (let i = 0; i < length; i++) {
            //length只是为了遍历完成当前层，和队列无关
            let cur = queue.shift()
            level.push(cur.val)
            //获取下一层节点
            if (cur.left) {
                queue.push(cur.left)
            }
            if (cur.right) {
                queue.push(cur.right)
            }
        }
        //放置当前层结果
        res.push(level)
    }
    return res
};

```
