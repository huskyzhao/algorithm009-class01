## 解法1



```javascript
var addDigits = function(num) {
    //终止条件
    if(num<10) return num
    //获取个位数的数组
    let nums = []
    while (num>=10) {
        let item = num%10
        nums.push(item)
        num = (num-item)/10
    }
    nums.push(num)
    let newNum = 0
    //获取个位数数组之和
    nums.forEach(element => {
        newNum+=element 
    });
    //递归
    return addDigits(newNum)
};

```
