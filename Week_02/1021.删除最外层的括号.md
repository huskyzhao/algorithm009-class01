# 1021.删除最外层的括号

## 解法1

时间复杂度O(n)
空间复杂度O(n)

```javascript
/*
 * @lc app=leetcode.cn id=1021 lang=javascript
 *
 * [1021] 删除最外层的括号
 */

// @lc code=start
/**
 * @param {string} S
 * @return {string}
 */
var removeOuterParentheses = function(S) {
    //设置一个栈
    let stack =[]
    //最终的结果
    let wantString = ""
    //每组的字符
    let groupString = ""
    for(let i = 0;i<S.length;i++){
        //判读后括号，处理栈信息
        if(S[i]==")"){
            stack.pop()
            //如果栈为空，则说明一个组数据已完成
            if(stack.length==0){
                wantString +=groupString.slice(1)
                groupString = ""
            }else{
                groupString += S[i]
            }
        }else{
            stack.push(S[i])
            groupString+=S[i]
        }
       
    }
    return wantString
};

```
