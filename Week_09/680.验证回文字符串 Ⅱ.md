# 解法一

时间复杂度O(n)
空间复杂度O(1)

```javascript
var validPalindrome = function(s) {
    let length = s.length
    if(length<=2) return true
    let start = 0
    let end = length-1
  
    while (start<end) {
        if(s[start]==s[end]){
            start++
            end--
            continue
        }else {
            return isPali(s,start,end-1)||isPali(s,start+1,end)
        }
      
    }
    return true
};
function isPali(str, l, r) { // 辅助函数
    while (l < r) {            // 指针相遇 结束循环
        if (str[l] !== str[r]) { // 一票否决
            return false;
        }
        l++;                     // 指针挪动，相互逼近
        r--;
    }
    return true;               // 没有遇到不同，返回true
}
```
